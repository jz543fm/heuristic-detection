/**
* Represents a function sleep.
* @function
* @constructor
* @param {any} milliseconds - Pauses execution time in ms
*/
function sleep(milliseconds) {
  var start = new Date().getTime();
  for (var i = 0; i < 1e7; i++) {
    if ((new Date().getTime() - start) > milliseconds){
      break;
    }
  }
}
